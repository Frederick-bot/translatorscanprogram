CREATE TABLE `user`       
(      
    `UID` INT(5) NOT NULL AUTO_INCREMENT ,      
    `username` VARCHAR(50) CHARACTER SET utf8 COLLATE utf8_general_ci NULL ,
    `password ` VARCHAR(50) CHARACTER SET utf8 COLLATE utf8_general_ci NULL ,
    PRIMARY KEY (`UID`)      
)       
ENGINE = InnoDB       
CHARSET=utf8       
COLLATE utf8_general_ci;
